### MYSQL
```sql
-- 课程类
CREATE TABLE COURSE(
  ID INT(3) NOT NULL PRIMARY KEY AUTO_INCREMENT,
  NAME VARCHAR(50) NOT NULL,
  TYPE INT(1) NOT NULL COMMENT '课程分类',
  CREDIT INT(2) NOT NULL COMMENT '学分',
  REMARK VARCHAR(400) DEFAULT NULL
);
-- 学生类
CREATE TABLE STUDENT(
  SNO VARCHAR(8) PRIMARY KEY COMMENT '年份(4位), 编号(4位)',
  CNO VARCHAR(7) NOT NULL COMMENT '班号',
  NAME VARCHAR(50) NOT NULL,
  GENDER ENUM('1','0') DEFAULT NULL COMMENT '1 : 男, 0 : 女',
  BIRTHDAY DATETIME(3) DEFAULT NULL,
  PHONE VARCHAR(11) DEFAULT NULL,
  REMARK VARCHAR(400) DEFAULT NULL
);
-- 教师类
CREATE TABLE TEACHER(
  TNO VARCHAR(9) PRIMARY KEY COMMENT '年份(4位) + 序号(4位)',
  NAME VARCHAR(50) NOT NULL,
  GENDER ENUM('1','0') DEFAULT NULL COMMENT '1 : 男, 0 : 女',
  BIRTHDAY DATETIME(3) DEFAULT NULL,
  PHONE VARCHAR(11) DEFAULT NULL,
  REMARK VARCHAR(400) DEFAULT NULL
);
-- 班号
CREATE TABLE CLASSES(
  CNO VARCHAR(7) PRIMARY KEY COMMENT '年份(4位) + 编号(3位)',
  MAIN_TNO VARCHAR(9) COMMENT '班主任编号',
  REMARK VARCHAR(400) DEFAULT NULL
);
-- 测试成绩
CREATE TABLE EXAM_RECORD(
  ID INT PRIMARY KEY AUTO_INCREMENT,
  SNO VARCHAR(9) NOT NULL,
  BATCHES_ID VARCHAR(20) NOT NULL,
  SCORE INT(3) NOT NULL
);
-- 考试批次
CREATE TABLE EXAM_BATCHES(
  ID VARCHAR(20) PRIMARY KEY COMMENT 'yyyyMMDD-hhmmss-xxxx',
  COURSE_ID INT(3) NOT NULL,
  NAME VARCHAR(50) NOT NULL,
  TOTAL_SCORE INT(3) NOT NULL,
  START_TIME DATETIME(3) NOT NULL,
  EXAM_TIEM int(3) NOT NULL COMMENT '分钟',
  REMARK VARCHAR(400) DEFAULT NULL
);
```
